// cpi.rs or utils.rs
pub fn settle_fill_cpi(
    risk_program:       &AccountView,
    user_account:       &AccountView,
    position:           &AccountView,
    market_config:      &AccountView,
    funding_state:      &AccountView,
    oracle:             &AccountView,
    fill:               &FillEvent,
    is_taker:           bool,
) -> ProgramResult {

    // serialize FillEvent as ix data
    let mut ix_data = [0u8; size_of::<FillEvent>() + 1];
    ix_data[0] = is_taker as u8;
    ix_data[1..].copy_from_slice(bytemuck::bytes_of(fill));

    pinocchio::cpi::invoke(
        &pinocchio::instruction::Instruction {
            program_id: risk_program.address(),
            accounts: &[
                user_account.to_meta_c(),
                position.to_meta_c(),
                market_config.to_meta_c(),
                funding_state.to_meta_c(),
                oracle.to_meta_c(),
            ],
            data: &ix_data,
        },
        &[
            user_account,
            position,
            market_config,
            funding_state,
            oracle,
        ],
    )
}
